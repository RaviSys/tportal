def index

  @program = Program.find(params[:program_id])

  if ActiveModel::Type::Boolean.new.cast(params[:calender])
    @lectures = @program.lectures
    calender_data
    render 'lecture_calender'
  else
    @lectures = @program.lectures.page(params[:page]).per(10)
  end
end
